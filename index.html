function submit() { const h = document.getElementById('hue').value; const s = document.getElementById('sat').value; const l = document.getElementById('light').value; // Reveal original color on the left half for direct comparison targetHalf.style.backgroundColor = `hsl(${target.h}, ${target.s}%, ${target.l}%)`; let hDiff = Math.abs(target.h - h); if (hDiff > 180) hDiff = 360 - hDiff; const sDiff = Math.abs(target.s - s); const lDiff = Math.abs(target.l - l); // Error calculation (Weighting Hue more heavily) const error = (hDiff / 180 * 7) + (sDiff / 100 * 2) + (lDiff / 100 * 2); // Scale score from 1 to 10 let roundScore = Math.max(1, 10 - error); roundScore = Math.round(roundScore * 10) / 10; history.push({ target: `hsl(${target.h}, ${target.s}%, ${target.l}%)`, user: `hsl(${h}, ${s}%, ${l}%)` }); totalScore += roundScore; scorePopup.innerText = roundScore.toFixed(1); mainBtn.innerText = (round < 5) ? "CONTINUE" : "FINISH"; mainBtn.onclick = (round < 5) ? nextRound : showEnd; }
